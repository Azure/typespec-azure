# A type referenced only through a paged result is still emitted

## TypeSpec

```tsp
@service(#{ title: "Random" })
namespace Random;

model ThingsListResult {
  @pageItems
  value: FakeData[];

  @nextLink
  nextLink?: string;

  moreStuff?: FakeInfo[];
}

model FakeData {
  someString: string;
}

model FakeInfo {
  someNumber: int32;
}

@list
@route("/things")
op listThings(): ThingsListResult;
```
