# A `@discriminator` model generates a polymorphic interface, subtypes, and unmarshal helpers, with sample generation

```yaml
generate-samples: true
```

## TypeSpec

```tsp
@armProviderNamespace
namespace Microsoft.Test;

union ProvisioningState {
  string,
  Succeeded: "Succeeded",
  Failed: "Failed",
  Canceled: "Canceled",
}

union PetKind {
  string,
  Cat: "Cat",
  Dog: "Dog",
}

#suppress "@azure-tools/typespec-azure-core/no-string-discriminator" "Existing"
@discriminator("kind")
model Pet {
  kind: PetKind;
  name: string;
}

model Cat extends Pet {
  kind: PetKind.Cat;
  purrs: boolean;
}

model Dog extends Pet {
  kind: PetKind.Dog;
  breed: string;
}

model PetResourceProperties {
  @visibility(Lifecycle.Read)
  provisioningState?: ProvisioningState;
}

model PetResource is TrackedResource<PetResourceProperties> {
  @path
  @key("petResourceName")
  @segment("petResources")
  name: string;
}

@armResourceOperations
interface Pets {
  get is ArmResourceRead<PetResource>;

  @post
  @armResourceAction(PetResource)
  adoptCat(...ResourceInstanceParameters<PetResource>, @bodyRoot cat: Cat):
    | ArmResponse<Cat>
    | ErrorResponse;

  @post
  @armResourceAction(PetResource)
  adoptPet(...ResourceInstanceParameters<PetResource>, @bodyRoot pet: Pet):
    | ArmResponse<Pet>
    | ErrorResponse;
}
```
