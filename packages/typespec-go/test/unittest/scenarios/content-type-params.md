# Content-type parameters are wired to the request body: optional bodies set it only when present, explicit unions select the literal

## TypeSpec

```tsp
@service(#{ title: "Regressions" })
namespace Regressions;

model SomeModel {
  name: string;
}

@route("/optional-binary-payload-with-content-type")
op optionalBinaryBodyWithContentType(
  @body payload?: bytes,
  @header("content-type") contentType?: string,
): void;

@route("/binary-payload-with-content-type")
op binaryBodyWithContentType(
  @body payload: bytes,
  @header("content-type") contentType: string,
): void;

@route("/payload-with-explicit-content-type")
op payloadWithExplicitContentType(
  @header("Content-Type")
  contentType: "application/snapshot+json" | "application/json",

  @body
  thing: SomeModel,
): void;
```
