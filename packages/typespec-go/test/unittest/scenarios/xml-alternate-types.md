# A `@@alternateType` swap combined with `@clientName` renames replaces the default XML wrapper models with Go-friendly arrays

## TypeSpec

```tsp
@service(#{ title: "Regressions" })
namespace Regressions {
  alias ContentXml = {
    @header("Content-Type")
    contentType: "application/xml";
  };

  model SignedIdentifiers {
    @Xml.unwrapped
    @Xml.name("SignedIdentifier")
    items: Array<SignedIdentifier>;
  }

  @Xml.name("SignedIdentifier")
  model SignedIdentifier {
    @Xml.name("Id") id: string;
  }

  @get
  @route("/xml-alternate-type-1")
  op getXmlOne(): ContentXml & SignedIdentifiers;

  @Xml.name("QueueMessagesList")
  model ReceivedMessages {
    @Xml.unwrapped
    items: Array<ReceivedMessage>;
  }

  @Xml.name("QueueMessage")
  model ReceivedMessage {
    @Xml.name("MessageId")
    messageId: string;
  }

  @get
  @route("/xml-alternate-type-2")
  op getXmlTwo(): ContentXml & ReceivedMessages;
}

namespace Customizations {
  model SignedIdentifiersGo is Array<Regressions.SignedIdentifier>;

  @clientName("Messages", "go")
  model ReceivedMessagesGo is Array<Regressions.ReceivedMessage>;
}

@@alternateType(Regressions.SignedIdentifiers, Customizations.SignedIdentifiersGo, "go");
@@clientName(Customizations.SignedIdentifiersGo, "SignedIdentifiers", "go");
@@alternateType(Regressions.ReceivedMessages, Customizations.ReceivedMessagesGo, "go");
@@clientName(Regressions.ReceivedMessage, "DequeuedMessage", "go");
```

	MessageID *string `xml:"MessageId"`
